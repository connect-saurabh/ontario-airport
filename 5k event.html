cost_data <- data.frame(
  category = c(
    "Signage",
    "Community Event Supplies/Vendors",
    "Volunteer Supplies",
    "Race Equipment/Supplies",
    "Timing Company/Race Management",
    "Event Equipment"
  ),
  total_cost = c(6054, 15211, 3646, 31883, 28478, 29686),
  cost_type = c(
    "Mixed",
    "Unrecoverable",
    "Unrecoverable",
    "Recoverable",
    "Recoverable",
    "Unrecoverable"
  )
)
revenue_data <- data.frame(
  source = c("Race Roster", "Sponsorships"),
  revenue = c(83802, 61000)
)
total_cost <- sum(cost_data$total_cost)
total_revenue <- sum(revenue_data$revenue)
net_profit <- total_revenue - total_cost
roi <- (net_profit / total_cost) * 100

recoverable_cost <- 62000
unrecoverable_cost <- 52958

scenarios <- data.frame(
  scenario = c("Low", "Baseline", "High"),
  revenue_multiplier = c(0.90, 1.00, 1.20)
)

scenario_results <- scenarios %>%
  mutate(
    projected_revenue = total_revenue * revenue_multiplier,
    projected_profit = projected_revenue - total_cost,
    projected_roi = (projected_profit / total_cost) * 100
  )

print(scenario_results)
  scenario revenue_multiplier projected_revenue projected_profit projected_roi

value_per_passenger <- 20

estimated_passengers_to_cover_total_cost <- total_cost / value_per_passenger
estimated_passengers_to_generate_net_profit <- net_profit / value_per_passenger

  metric = c(

    "Passenger Equivalent of Net Profit"
  ),
  value = c(

    estimated_passengers_to_generate_net_profit
