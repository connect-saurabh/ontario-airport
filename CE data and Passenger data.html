Merge Event Data with Passenger Data

data_airline <- passengers_airline %>%
  left_join(events_monthly, by = c("year", "month")) %>%
  mutate(
    total_events = ifelse(is.na(total_events), 0, total_events),
    season = case_when(
      month %in% c(12, 1, 2) ~ "Winter",
      month %in% c(3, 4, 5) ~ "Spring",
      month %in% c(6, 7, 8) ~ "Summer",
      month %in% c(9, 10, 11) ~ "Fall"
    ),
    season = factor(season, levels = c("Winter", "Spring", "Summer", "Fall"))

  arrange(year, month, airline)

Simulate Summer +20% Event Scenario

scenario_summer <- data_airline %>%
  mutate(
    simulated_events = ifelse(season == "Summer", total_events * 1.20, total_events)
  )

scenario_summer$scenario_pred <- predict(
  model_sim,
  newdata = scenario_summer %>% mutate(total_events = simulated_events)
)

scenario_summer <- scenario_summer %>%
  mutate(
    incremental_passengers = scenario_pred - baseline_pred,
    scenario = "Summer +20%"
  )

scenarios <- data.frame(
  scenario = c("Baseline", "+10%", "+20%", "+30%"),
  multiplier = c(1.00, 1.10, 1.20, 1.30)
)
scenario_results <- lapply(1:nrow(scenarios), function(i) {

  mult <- scenarios$multiplier[i]
  scen_name <- scenarios$scenario[i]

  temp <- data_airline %>%
    mutate(
      simulated_events = total_events * mult
    )

  temp$predicted <- predict(
    model_sim,
    newdata = temp %>% mutate(total_events = simulated_events)
  )

  temp <- temp %>%
    mutate(
      baseline = predict(model_sim, newdata = data_airline),
      incremental_passengers = predicted - baseline,
      scenario = scen_name
    )

  return(temp)
})

scenario_results <- bind_rows(scenario_results)
scenario_summary <- scenario_results %>%
  group_by(scenario) %>%
  summarise(
    total_incremental_passengers = sum(incremental_passengers, na.rm = TRUE),
    avg_incremental_passengers = mean(incremental_passengers, na.rm = TRUE),
    .groups = "drop"
  )
